(function () {
    "use strict";
 
    var app = angular.module("todo");
 
    app.controller("TodoController", ["$window", "$location", "Auth", "Todo", TodoController]);
 
    function TodoController($window, $location, Auth, Todo) {
      var ctrl = this;
 
      // ctrl.items = [
      //   {id: 1, title: 'aaa', status: 'in-progress'},
      //   {id: 2, title: 'bbb', status: 'in-progress'},
      //   {id: 3, title: 'ccc', status: 'in-progress'},
      //   {id: 4, title: 'ddd', status: 'in-progress'},
      //   {id: 5, title: 'xxx', status: 'completed'},
      //   {id: 6, title: 'yyy', status: 'completed'},
      //   {id: 7, title: 'zzz', status: 'completed'}
      // ];
 
      ctrl.handleDrop = handleDrop;
 
      init();
 
      function init() {
        var user = Auth.getUser();
        Todo.getAll(user.sessionId)
          .then(function(data) {
            if (data && data.status === 'error') {
              Auth.logout();
              //$window.localStorage.removeItem('user');
              $location('/');
              return;
            }
            ctrl.items = data.data;
          })
          .catch(function(data) {
            Auth.logout();
            $location.path('/');
            return;
          });
      }
 
      function handleDrop(id, containerSelectorId) {
        console.log('Item ' + id + ' dropped on ' + containerSelectorId);
      }
    }
})();
